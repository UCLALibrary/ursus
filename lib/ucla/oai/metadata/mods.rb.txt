# frozen_string_literal: true
module Ucla::Oai::Metadata
    # Simple implementation of the DPLA metadata format.
    class Mods < OAI::Provider::Metadata::Format
      def initialize
        @prefix = 'oai_mods'
        @element_namespace = 'mods'
        @namespace = 'http://www.loc.gov/mods/v3'
        @schema = 'https://www.loc.gov/standards/mods/v3/mods-3-8.xsd'
      end
  
      def header_specification
        {
          'xmlns:oai_dpla' =>  "https://digital.library.ucla.edu/oai_dpla/",
          'xmlns:dc' => "http://purl.org/dc/elements/1.1/",
          'xmlns:dcterms' => "http://purl.org/dc/terms/",
          'xmlns:edm' => "http://www.europeana.eu/schemas/edm/",
          'xmlns:xsi' => "http://www.w3.org/2001/XMLSchema-instance",
          'xsi:schemaLocation' =>
            %(https://digital.library.ucla.edu/oai_dpla/ https://digital.library.ucla.edu/oai_dpla/oai_dpla.xsd).gsub(/\s+/, ' ')
        }
      end
    end
  end
